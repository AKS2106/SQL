Step 6 Write the Glue ETL script and upload to S3 ▼
Create src\glue_transform.py on your personal laptop
import sys
from awsglue.transforms import *
from awsglue.utils import getResolvedOptions
from pyspark.context import SparkContext
from awsglue.context import GlueContext
from awsglue.job import Job
from awsglue.dynamicframe import DynamicFrame
from pyspark.sql.functions import (
    col, when, avg, round as round_,
    count, lit, regexp_replace, trim
)
from pyspark.sql.window import Window

# ── Glue boilerplate ─────────────────────────────────────────────
args = getResolvedOptions(sys.argv, ["JOB_NAME"])
sc   = SparkContext()
glueContext = GlueContext(sc)
spark = glueContext.spark_session
job  = Job(glueContext)
job.init(args["JOB_NAME"], args)
spark.sparkContext.setLogLevel("ERROR")

BUCKET   = "YOUR-NAME-de-auto-pipeline"   # ← replace with your bucket
OUT_PATH = f"s3://{BUCKET}/processed/vehicles/"

print("=" * 55)
print("  AUTO PIPELINE — GLUE ETL JOB")
print("=" * 55)

# ── Step 1: Read from Glue Catalog ───────────────────────────────
print("\n[1/5] Reading from Glue Catalog...")
dyf = glueContext.create_dynamic_frame.from_catalog(
    database           = "auto_pipeline_db",
    table_name         = "vehicles",
    transformation_ctx = "raw_vehicles"
)
df = dyf.toDF()
print(f"  Loaded: {df.count():,} rows, {len(df.columns)} columns")

# ── Step 2: Clean and validate ───────────────────────────────────
print("\n[2/5] Cleaning...")

df = df \
    .filter(col("price").isNotNull() & (col("price") > 0)) \
    .filter(col("year").isNotNull()  & (col("year")  > 1990)) \
    .filter(col("odometer").isNotNull()) \
    .withColumn("price",    col("price").cast("double")) \
    .withColumn("year",     col("year").cast("integer")) \
    .withColumn("odometer", col("odometer").cast("double")) \
    .withColumn("manufacturer",
        trim(regexp_replace(col("manufacturer"), "[^a-zA-Z ]", "")))

print(f"  After cleaning: {df.count():,} rows")

# ── Step 3: Add business features ────────────────────────────────
print("\n[3/5] Adding features...")

# Vehicle age
from pyspark.sql.functions import lit as lit_
df = df.withColumn("vehicle_age",
    lit_(2024).cast("integer") - col("year"))

# Price tier
df = df.withColumn("price_tier",
    when(col("price") <  5000,  "Budget")
    .when(col("price") < 15000, "Economy")
    .when(col("price") < 30000, "Mid-range")
    .otherwise("Premium"))

# Mileage tier
df = df.withColumn("mileage_tier",
    when(col("odometer") <  30000, "Low")
    .when(col("odometer") < 80000, "Medium")
    .when(col("odometer") < 150000,"High")
    .otherwise("Very High"))

# Price per mile (value metric)
df = df.withColumn("price_per_mile",
    when(col("odometer") > 0,
         round_(col("price") / col("odometer"), 4))
    .otherwise(None))

# Compare to segment average — is listing underpriced?
w_segment = Window.partitionBy("manufacturer","condition")
df = df \
    .withColumn("avg_price_segment",
        round_(avg("price").over(w_segment), 2)) \
    .withColumn("vs_segment_avg",
        round_(col("price") - col("avg_price_segment"), 2)) \
    .withColumn("is_underpriced",
        when(col("vs_segment_avg") < -2000, True)
        .otherwise(False))

# ── Step 4: Summary stats ────────────────────────────────────────
print("\n[4/5] Summary by price tier:")
df.groupBy("price_tier") \
  .agg(
      count("*").alias("listings"),
      round_(avg("price"), 0).alias("avg_price"),
      round_(avg("vehicle_age"), 1).alias("avg_age_yrs")
  ).orderBy("avg_price").show()

print("\nTop 5 manufacturers by listing count:")
df.groupBy("manufacturer") \
  .count() \
  .orderBy(col("count").desc()) \
  .limit(5).show()

# ── Step 5: Write to S3 as Parquet ───────────────────────────────
print(f"\n[5/5] Writing to S3: {OUT_PATH}")
df.coalesce(8) \
  .write \
  .mode("overwrite") \
  .partitionBy("price_tier") \
  .parquet(OUT_PATH)

print("\n✓ ETL complete")
job.commit()   # CRITICAL — advances Job Bookmarks

-- Set database first in Athena console dropdown: auto_pipeline_db

-- Q1: Listing count and average price by price tier
SELECT
    price_tier,
    COUNT(*)                              AS total_listings,
    ROUND(AVG(price), 0)                  AS avg_price,
    ROUND(AVG(vehicle_age), 1)            AS avg_age_years,
    ROUND(AVG(odometer)/1000, 0)          AS avg_miles_k
FROM vehicles
GROUP BY price_tier
ORDER BY avg_price;

-- Q2: Top 10 manufacturers by number of listings
SELECT
    manufacturer,
    COUNT(*)                              AS listings,
    ROUND(AVG(price), 0)                  AS avg_price,
    ROUND(AVG(odometer)/1000, 0)          AS avg_miles_k
FROM vehicles
WHERE manufacturer IS NOT NULL
GROUP BY manufacturer
ORDER BY listings DESC
LIMIT 10;

-- Q3: Underpriced listings — potential deals
-- Cars priced more than $2000 below their segment average
SELECT
    manufacturer,
    model,
    year,
    price,
    avg_price_segment,
    vs_segment_avg,
    condition,
    odometer,
    region
FROM vehicles
WHERE is_underpriced = true
    AND condition IN ('good','excellent','like new')
ORDER BY vs_segment_avg ASC
LIMIT 20;

-- Q4: Price by condition — how condition affects value
SELECT
    condition,
    COUNT(*)                              AS listings,
    ROUND(AVG(price), 0)                  AS avg_price,
    ROUND(MIN(price), 0)                  AS min_price,
    ROUND(MAX(price), 0)                  AS max_price
FROM vehicles
WHERE condition IS NOT NULL
GROUP BY condition
ORDER BY avg_price DESC;

-- Q5: Window function — rank manufacturers by avg price within region
-- Shows top 3 most expensive brands per region
WITH ranked AS (
    SELECT
        region,
        manufacturer,
        ROUND(AVG(price), 0)              AS avg_price,
        COUNT(*)                          AS listings,
        DENSE_RANK() OVER (
            PARTITION BY region
            ORDER BY AVG(price) DESC
        )                                 AS price_rank
    FROM vehicles
    WHERE manufacturer IS NOT NULL
        AND region IS NOT NULL
    GROUP BY region, manufacturer
    HAVING COUNT(*) >= 10
)
SELECT * FROM ranked
WHERE price_rank <= 3
ORDER BY region, price_rank;

# Automobile Sales Pipeline — AWS End-to-End

End-to-end data pipeline on AWS processing 426,880 vehicle
listings. Raw CSV → S3 → Glue Crawler → Glue ETL (PySpark)
→ Parquet on S3 → Athena SQL queries.

## Business Problem
A used car marketplace needs to identify underpriced listings,
track regional pricing trends, and generate weekly sales reports.
Manual analysis took hours — this pipeline produces insights in
minutes on demand.

## Architecture
Raw CSV → S3 (raw zone) → Glue Crawler (schema detection)
→ Glue Catalog → Glue ETL Job (PySpark transform)
→ S3 (processed Parquet, partitioned by price_tier)
→ Athena (SQL queries) → Business insights

## AWS Services Used
S3, Glue Crawler, Glue Catalog, Glue ETL, Athena, IAM

## Key Results
- Records processed: 426,880 listings
- Underpriced listings identified: [X] (priced $2000+ below segment avg)
- Athena data scanned for Q1: [X MB] (partition pruning working)
- Glue job duration: [X minutes]

## Key Technical Decisions
- Parquet + partitionBy(price_tier): downstream Athena queries
  filter by price tier — partition pruning reduces scan cost
- coalesce(8): prevents small files problem from Glue defaults
- Window function for segment avg: AVG OVER (PARTITION BY
  manufacturer, condition) — compares each listing to its
  direct competitors without collapsing the DataFrame
- job.commit(): ensures Job Bookmarks advance so reruns only
  process new files — not the full dataset again

## How to Run
1. Configure AWS CLI: aws configure
2. Create S3 bucket and upload CSV (see Step 3 in setup guide)
3. Run Glue Crawler to detect schema
4. Upload script: aws s3 cp src/glue_transform.py s3://BUCKET/scripts/
5. Run Glue ETL job from AWS Console
6. Query results in Athena using sql/analysis.sql

## Skills Demonstrated
AWS S3, Glue, Athena, IAM | PySpark on managed Spark cluster
Window functions | Partition pruning | Cloud ETL pipeline design
